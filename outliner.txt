Rough UML Class Outline

@startuml


enum Gender {
    FEMALE
    MALE
    UNKNOWN
}

enum BodyType {
    AVERAGE
    ATHLETIC
    OVERWEIGHT
    UNSPECIFIED
}

enum AgeCategory {
    BABY
    CHILD
    ADULT
    SENIOR
}

enum Profession {
    DOCTOR
    CEO
    CRIMINAL
    HOMELESS
    UNEMPLOYED
    NONE
}

class Character {
    - age: int
    - gender: Gender
    - bodyType: BodyType
}

class Human extends Character{
    - profession: Profession
    - isPregnant: boolean
    - ageCategory: AgeCategory
}

class Animal extends Character{
    - species: String
    - isPet: boolean
}

class Location {
    + boolean isTrespassing
    + List<Character> characters
}

class Scenario {
    + String emergencyDetails
    + List<Location> locations
}

interface DecisionAlgorithm {
    + decide(Scenario): Location
}

class User {
    + List<Decision> decisions
    + Statistics statistics
}

class RescueBot {
    + DecisionAlgorithm decisionAlgorithm
    + rescue(Scenario): void
}

class Statistics {
    + Map<String, Double> detailsPercentage
}

class AuditLog {
    + String outputFile
    + log(Decision): void
}

class ScenarioManager {
    + List<Scenario> scenarios
    + loadScenarios(String): void
    + generateScenario(): Scenario
}

class Decision {
    + User user
    + Scenario scenario
    + Location decision
}

@enduml
